```python
class Database:
    def __init__(self, db_path: str = "clay_dupe.db"):
        self.db_path = db_path
        self._conn: Optional[aiosqlite.Connection] = None
        self._init_db()

    @asynccontextmanager
    async def _connect(self):
        conn = await self._get_connection()
        try:
            yield conn
            await conn.commit()
        except Exception:
            await conn.rollback()
            raise
```
