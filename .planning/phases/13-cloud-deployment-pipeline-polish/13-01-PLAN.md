Create `entrypoint.sh`:
```bash
#!/bin/bash
set -e

# Ensure data directory exists and is writable
mkdir -p /data
if [ ! -w /data ]; then
    echo "ERROR: /data is not writable" >&2
    exit 1
fi

# Start Streamlit (exec replaces shell for proper signal handling)
exec streamlit run ui/app.py \
    --server.port="${PORT:-8501}" \
    --server.address=0.0.0.0 \
    --server.headless=true \
    --browser.gatherUsageStats=false
```

Extend `.github/workflows/ci.yml` to add a Docker build verification job:

```yaml
  docker-build:
    runs-on: ubuntu-latest
    needs: test
    if: github.ref == 'refs/heads/main'

    steps:
      - uses: actions/checkout@v4

      - name: Build Docker image
        run: docker build -t clay-dupe:${{ github.sha }} .

      - name: Verify container starts and health check passes
        run: |
          docker run -d --name test-container -p 8501:8501 \
            -e APP_PASSWORD=test123 \
            clay-dupe:${{ github.sha }}
          sleep 15
          curl --fail http://localhost:8501/_stcore/health || exit 1
          docker stop test-container
```

This job runs only on pushes to main, after tests pass. Railway's native GitHub integration handles actual deployment (auto-deploys on push to main). No `railway up` CLI step needed.
